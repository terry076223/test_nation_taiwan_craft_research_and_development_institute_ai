▶ 展開 Google Apps Script 程式碼（複製後貼入 GAS 編輯器）
步驟：前往 script.google.com → 新增專案 → 貼上以下程式碼 → 部署為 Web App（執行身分：我、存取權限：所有人）→ 複製部署網址
function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName('回饋記錄') || ss.getActiveSheet();
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['時間', '機器人', '使用者問題', 'AI回覆', '評分']);
  }
  var data = JSON.parse(e.postData.contents);
  sheet.appendRow([
    Utilities.formatDate(new Date(), 'Asia/Taipei', 'yyyy/MM/dd HH:mm:ss'),
    data.botName || '',
    data.question || '',
    data.answer || '',
    data.rating === 1 ? '👍 滿意' : '👎 不滿意'
  ]);
  return ContentService
    .createTextOutput(JSON.stringify({status:'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName('回饋記錄') || ss.getActiveSheet();
  var lastRow = sheet.getLastRow();
  if (lastRow <= 1) {
    return ContentService.createTextOutput('[]').setMimeType(ContentService.MimeType.JSON);
  }
  var rows = sheet.getRange(2, 1, lastRow - 1, 5).getValues().map(function(r) {
    return { dateStr: r[0], botName: r[1], question: r[2], answer: r[3],
             ratingText: r[4], rating: String(r[4]).indexOf('👍') >= 0 ? 1 : -1 };
  }).reverse();
  return ContentService
    .createTextOutput(JSON.stringify(rows))
    .setMimeType(ContentService.MimeType.JSON);
}
貼上部署網址後按「儲存」，之後所有回饋都會寫入 Google Sheets
